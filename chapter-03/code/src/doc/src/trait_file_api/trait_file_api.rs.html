1//! Trait-based file API example that moves read behavior behind a trait.
2
3trait Read {
4    fn read(&self, save_to: &mut Vec<u8>) -> Result<usize, String>;
5}
6
7/// Represents an in-memory file whose bytes can be read through the `Read` trait.
8#[derive(Debug)]
9pub struct File {
10    name: String,
11    data: Vec<u8>,
12}
13
14impl File {
15    fn new(name: &str) -> File {
16        File {
17            name: String::from(name),
18            data: Vec::new(),
19        }
20    }
21
22    fn new_with_data(name: &str, data: &Vec<u8>) -> File {
23        let mut f = File::new(name);
24        f.data = data.clone();
25        f
26    }
27}
28
29impl Read for File {
30    fn read(&self, save_to: &mut Vec<u8>) -> Result<usize, String> {
31        let mut tmp = self.data.clone();
32        let read_length = tmp.len();
33
34        save_to.reserve(read_length);
35        save_to.append(&mut tmp);
36        Ok(read_length)
37    }
38}
39
40pub fn trait_file_api() {
41    let f = File::new_with_data("f3.txt", &vec![114, 117, 115, 116, 33]);
42    let mut buffer = vec![];
43    let n_bytes = f.read(&mut buffer).unwrap();
44    println!("{} byte(s) read from {:?}", n_bytes, f);
45}
